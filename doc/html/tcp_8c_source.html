388 static bool TCP_Server_ConnectNewClient(TCP_S_t* _TCP)
389 {
390  if (! IsStructValid(_TCP) )
391  {
392  return FALSE;
393  }
394 
395  struct sockaddr_in sIn;
396  memset(&sIn , 0 , sizeof(sIn) );
397  uint addr_len;
398  addr_len = sizeof(sIn);
399  int socket;
400  SocketInfo_t* aSI;
401 
402  socket = accept(_TCP->m_listenSocket, (struct sockaddr *) &sIn, &addr_len ) ;
403 
404  /* got real new socket but over capacity, so close connection */
405  if (0 < socket && _TCP->m_connectionCapacity <= _TCP->m_connectedNum)
406  {
407  #if !defined(NDEBUG) /* DEBUG */
408  printf("server has too many (%d) connection. dropping new socket #%d.\n", _TCP->m_connectedNum, socket);
409  #endif
410 
411  if(_TCP->m_errorFunc)
412  {
413  _TCP->m_errorFunc(TOO_MANY_CONNECTION, socket, NULL);
414  }
415 
416  close(socket);
417  return FALSE;
418  }
419 
420  if (socket > 0)
421  { /* Success accept link */
422 
423  if (IS_NON_BLOCKING_METHOD == TRUE)
424  {
425  SetSocketBlockingEnabled(socket, FALSE);
426  }
427 
428 
429  /* add new socket to list of sockets */
430  aSI = CreateSocketInfo(socket, _TCP->m_timeoutMS);
431  if (!aSI)
432  {
433  close(socket);
434  return FALSE;
435  }
436 
437  if (! list_rpush(_TCP->m_sockets, list_node_new(aSI)) )
438  {
439  /* check list fail */
440  close(socket);
441  DestorySocketInfo(aSI);
442  return FALSE;
443  }
444 
445  _TCP->m_connectedNum++;
446 
447  if (_TCP->m_newConnectionFunc)
448  {
449  /* if user provide a function to invoke when new client connected */
450  _TCP->m_newConnectionFunc(socket, _TCP->m_contex);
451  }
452 
453  return TRUE;
454  }
455  else if ( IsFail_nonBlocking( socket) )
456  { /* Failed accept link */
457  perror("TCP_ServerConnect accept Failed.");
458  return FALSE;
459  }
460  else
461  { /* no one on the other side, all is well */
462  return FALSE;
463  }
